Setup Instructions

  1. Create a folder named "Website Images" in your Google Drive
  2. Upload some images to this folder
  3. Create a Google Apps Script with the code below
  4. Deploy it as a web app with "Execute as me" and "Anyone" access
  5. Replace the SCRIPT_URL in the JavaScript code with your deployment URL

Google Apps Script Code:

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const userEmail = data.email;
    
    // Get the Drive folder named "Website Images"
    var folderIterator = DriveApp.getFoldersByName("Website Images");
    if (!folderIterator.hasNext()) {
      return ContentService.createTextOutput(JSON.stringify({
        result: 'error', 
        message: 'Folder "Website Images" not found. Please create it and add some images.'
      })).setMimeType(ContentService.MimeType.JSON);
    }
    
    var folder = folderIterator.next();
    var files = folder.getFiles();
    var images = [];
    
    // Process image files
    while (files.hasNext()) {
      var file = files.next();
      var mimeType = file.getMimeType();
      
      // Check if file is an image
      if (mimeType.indexOf('image/') !== -1) {
        images.push({
          id: file.getId(),
          name: file.getName(),
          url: 'https://drive.google.com/uc?id=' + file.getId(),
          date: file.getDateCreated().toISOString().split('T')[0],
          size: Math.round(file.getSize() / 1024) + ' KB'
        });
      }
    }
    
    if (images.length === 0) {
      return ContentService.createTextOutput(JSON.stringify({
        result: 'error', 
        message: 'No images found in the "Website Images" folder.'
      })).setMimeType(ContentService.MimeType.JSON);
    }
    
    // Send confirmation email
    try {
      MailApp.sendEmail({
        to: userEmail,
        subject: 'Google Drive Image Access',
        htmlBody: `Hello,

You recently accessed your Google Drive images from our website. ${images.length} images were retrieved from your "Website Images" folder.`
      });
    } catch (emailError) {
      console.error('Error sending email:', emailError);
    }
    
    return ContentService.createTextOutput(JSON.stringify({
      result: 'success',
      images: images
    })).setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({
      result: 'error', 
      message: error.toString()
    })).setMimeType(ContentService.MimeType.JSON);
  }
}
