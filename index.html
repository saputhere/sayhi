Send "Hello World" to cholee.lk@gmail.com

Enter your email below and press the button. It will send "Hello World" automatically via Google Apps Script.

Setup required:
  1. Go to Google Apps Script.
  2. Create a new project, paste the script shown below, and deploy as a Web App with access set to "Anyone".
  3. Copy the Web App URL and replace YOUR_WEB_APP_URL in the script.
function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var to = "cholee.lk@gmail.com";
    var subject = "Message from Website";
    var body = "Hello World\nFrom: " + data.from;
    MailApp.sendEmail(to, subject, body);
    return ContentService.createTextOutput("OK");
  } catch(err){
    return ContentService.createTextOutput("Error: " + err.message);
  }
}
